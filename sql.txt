CREATE TABLE  "LOGIN" 
   (	"TID" NUMBER(*,0) NOT NULL ENABLE, 
	"USERNAME" VARCHAR2(20) NOT NULL ENABLE, 
	"PASSWORD" VARCHAR2(30) NOT NULL ENABLE, 
	 FOREIGN KEY ("TID")
	  REFERENCES  "TRADER" ("TID") ENABLE
   )
CREATE TABLE  "TRADER" 
   (	"TID" NUMBER(*,0) NOT NULL ENABLE, 
	"FNAME" VARCHAR2(30) NOT NULL ENABLE, 
	"LNAME" VARCHAR2(30) NOT NULL ENABLE, 
	"EMAIL" VARCHAR2(30) NOT NULL ENABLE, 
	"ROLE" VARCHAR2(20) NOT NULL ENABLE, 
	"BALANCE" NUMBER(20,0), 
	 PRIMARY KEY ("TID") ENABLE
   )
CREATE TABLE  "TRANSACTION" 
   (	"TRANS_ID" NUMBER(*,0) NOT NULL ENABLE, 
	"TID" NUMBER(*,0) NOT NULL ENABLE, 
	"S_NAME" VARCHAR2(20) NOT NULL ENABLE, 
	"QTY" NUMBER(*,0) NOT NULL ENABLE, 
	"STATUS" VARCHAR2(10) NOT NULL ENABLE, 
	"T_DATE" VARCHAR2(30) NOT NULL ENABLE, 
	"PROCESS" VARCHAR2(10) NOT NULL ENABLE, 
	 PRIMARY KEY ("TRANS_ID") ENABLE, 
	 FOREIGN KEY ("TID")
	  REFERENCES  "TRADER" ("TID") ENABLE
   )
